(a) i = 1;
    printf("%d ", i++ - 1);
    printf("%d", i);

    output:
    0 1

(b) i = 10; j = 5;
    printf("%d ", i++ - ++j);
    printf("%d %d", i, j);

    output:
    4 11 6

(c) i = 7; j = 8;
    printf("%d ", i++ - --j);
    printf(%d %d", i, j);

    output:
    0 8 7

(d) i = 3; j = 4; k = 5;
    printf("%d ", i++ - j++ + --k);
    printf("%d %d %d", i, j, k);

    output:
    3 4 5 4
