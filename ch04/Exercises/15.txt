(a) i += j
    i = i + j
    i = 1 + 2
    i = 3

    i = 3, j = 2

(b) i--;
    i = i - 1;
    i = 3 - 1;
    i = 2;
    
    i = 2, j = 2

(c) i * j / i
    (2 * 2) / 2
    4 / 2
    2
    No value assigned to i or j

    i = 2, j = 2

(d) i % ++j
    i % (++j)
    2 % (j = 2 + 1)
    2 % (j = 3)
    2 % 3
    2
    No value assigned to i

    i = 2, j = 3
